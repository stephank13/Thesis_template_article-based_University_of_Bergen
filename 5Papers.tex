\newcommand{\watermark}[1]{\tikz[remember picture, overlay] \node [red,opacity=0.3,rotate=45,scale=5.0] at (current page.center) {#1};}

% commands and settings for thumb indices created with \fancytab command
\newcommand{\nop}{3}                                                % put in the number of papers you have in your thesis
\newcommand{\rtw}{.12}                                              % relative width of thumb index (0.18 is used by printing company)
\newcommand{\thumbheight}{\dimexpr\paperheight/\nop\relax}
\FPeval{\rmpw}{1-\rtw}
\fancytabsStyle{\Huge\scshape}
\fancytabsHeight{\thumbheight}
\fancytabsWidth{\rtw\paperwidth}
\fancytabsCount{\nop}
\fancytabsLeftColor{black}
\fancytabsRightColor{black} 
\fancytabsTop{0mm}
% \fancytabsTextVPos{0.5} 
% \fancytabsTextHPos
% \fancytabsGap 
%\fancytabsFloor 
\fancytabsRotate{90}                    % change to 0 for non rotated index 
\newcommand{\thumbindex}[2]{% 
    \let\eastwest\undefined 
    \let\multiplier\undefined 
    \fancytab[RIGHT]{\scalebox{1.5}{\textbf{\textcolor{white}{#1}}}}{#2}
    }
% \thumbindex requires two inputs, text and a position index (numer between 1 and \nop (number of papers)
